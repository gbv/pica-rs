pub fn into_owned(self) -> FieldMut

Converts the immutable tag into its mutable counterpart by consuming the source.

Example
use pica_record::TagRef;

fn example() -> anyhow::Result<()> {
    let tag = TagRef::new("003@").into_owned();
    assert_eq!(tag, "003@");
    Ok(())
}
